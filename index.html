export const getRates = async (startAfter, orderBy, pagesize = 10) => {
    let query =
        firebase.firestore().collection('rates')
            .orderBy(orderBy, 'desc')
            .limit(pagesize)
    if (startAfter !== null) {
        query = query.startAfter(startAfter)
    }
    return await query.get()
}
